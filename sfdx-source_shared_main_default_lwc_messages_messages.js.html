    import {LightningElement, api, track} from 'lwc';
import {Copy} from "c/utils";

/**
 * Enum for ApexPages.Severity values to represent in JavaScript.
 * @readonly
 * @enum
 */
const Severity = {
	/** @type String */
	SUCCESS: `SUCCESS`,
	/** @type String */
	INFO: `INFO`,
	/** @type String */
	WARNING: `WARNING`,
	/** @type String */
	ERROR: `ERROR`
};
Object.freeze(Severity);

/**
 *
 */
class Messages extends LightningElement {
	/**
	 * @typedef Message
	 * @property {String} title
	 * @property {Severity} severity
	 * @property {Number} duration
	 */
	
	
	@api
	get messages() {
		if (!this.__messages) {
			this.__messages = [];
		}
		return this.__messages;
	};
	
	set messages(value) {
		if (!value) {
			this.__messages = [];
			return;
		}
		
		this.__messages = Copy.deep(value);
		
		this.prepareMessages();
	}
	
	prepareMessages() {
		this.__messages.forEach(message => {
			Messages.setMessageKey(message);
			
			message.classes =
				`slds-notify slds-notify_${this.type} slds-theme_${this.type}-texture `;
			if (message.severity === Severity.SUCCESS) {
				message.classes += `slds-theme_success`;
				message.icon = 'utility:success';
				message.variant = 'inverse';
				message.buttonVariant = 'border-inverse';
			} else if (message.severity === Severity.INFO) {
				message.classes += `slds-theme_info`;
				message.icon = 'utility:info';
				message.variant = 'inverse';
				message.buttonVariant = 'border-inverse';
			} else if (message.severity === Severity.WARNING) {
				message.classes += `slds-theme_warning`;
				message.icon = 'utility:warning';
				message.variant = 'warning';
				message.buttonVariant = 'warning';
			} else if (message.severity === Severity.ERROR) {
				message.classes += `slds-theme_error`;
				message.icon = 'utility:error';
				message.variant = 'inverse';
				message.buttonVariant = 'border-inverse';
			}
			
			if (message.duration) {
				setTimeout(() => this.closeMessage(message.key), message.duration);
			}
		});
		
		this.hasMessages = this.__messages.length > 0;
		
		this.containerClass = this.type === 'alert' ? '' : 'message-toasts slds-notify_container';
	}
	
	static setMessageKey(message) {
		message.key = message.key ?? Messages.__counter++;
	}
	
	@api
	get type() {
		return this.__type || 'alert';
	}
	
	set type(value) {
		this.__type = value;
		
		this.prepareMessages();
	}
	
	@track hasMessages = false;
	@track __messages = [];
	
	static __counter = 0;
	
	@track containerClass = '';
	
	close(event) {
		let messageKey = parseInt(event.target.dataset.messageKey);
		this.closeMessage(messageKey);
	}
	
	closeMessage(messageKey) {
		const message = this.__messages.find(message => message.key === messageKey);
		if (message) {
			this.dispatchEvent(new CustomEvent('closemessage', {detail: message, bubbles: true}));
			
			this.__messages = this.__messages.filter(message => message.key !== messageKey);
		}
	}
}

export default Messages;
