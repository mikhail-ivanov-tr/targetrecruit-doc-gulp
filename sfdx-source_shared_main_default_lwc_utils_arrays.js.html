    export class Arrays {
	static isEmpty(array) {
		return !this.isNotEmpty(array);
	}
	
	static isNotEmpty(array) {
		return array && array.length > 0;
	}
	
	/**
	 * Splits an array to chunks.
	 * @param array Array to split.
	 * @param size Size of each chunk.
	 * @returns {*}
	 */
	static chunk(array, size) {
		return array.reduce((resultArray, item, index) => {
			const chunkIndex = Math.floor(index / size);
			
			if (!resultArray[chunkIndex]) {
				resultArray[chunkIndex] = []; // start a new chunk
			}
			
			resultArray[chunkIndex].push(item);
			
			return resultArray;
		}, []);
	}
	
	static sort(array, sortBy, sortDirection = `asc`) {
		return array.sort((item1, item2) => {
			let value1 = typeof sortBy === 'function' ? sortBy(item1) : item1[sortBy];
			let value2 = typeof sortBy === 'function' ? sortBy(item2) : item2[sortBy];
			return sortDirection === `asc` ? value1 > value2 ? 1 : -1 : value1 > value2 ? -1 : 1;
		});
	}
	
	static unique(array, fieldName) {
		return Array.from(new Set(fieldName ? array.map(item => item[fieldName]) : array));
	}
	
	static groupBy(array, uniquenessCallback) {
		const result = {};
		for (const item of array) {
			const uniquenessKey = uniquenessCallback(item);
			let keyArray = result[uniquenessKey];
			if (!keyArray) {
				keyArray = [];
			}
			
			keyArray.push(item);
			
			result[uniquenessKey] = keyArray;
		}
		
		return result;
	}
	
	static toObject(array, key, value) {
		return array.reduce((result, item) => {return {...result, [item[key]]: item[value],};}, {});
	}
	
	static remove(array, compareFunc) {
		while (true) {
			let value = array.find(compareFunc);
			if (!value) {
				return;
			}
			
			const index = array.indexOf(value);
			if (index > -1) {
				array.splice(index, 1);
			} else {
				return;
			}
		}
	}
	
	static removeByIndex(array, index) {
		array.splice(index, 1);
	}
	
	static addAll(arrayTarget, arraySource) {
		return arrayTarget.push(...arraySource);
	}
}
